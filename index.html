import pandas as pd import json # Load the CSV df = pd.read_csv('Client Map - Sheet1.csv', skiprows=1) # Clean column names df.columns = [c.strip() for c in df.columns] # Filter out empty rows where Customer is missing df = df[df['Customer'].notna()] clients_list = [] for i, row in df.iterrows(): scope = [] # Logic for scope: if 'x' just add category, if text add "Category: Text" for cat in ['Social', 'PPC', 'Creative', 'Ad hoc']: val = str(row[cat]).strip() if pd.notna(row[cat]) else "" if val.lower() == 'x': scope.append(cat) elif val != "" and val.lower() != 'nan': scope.append(f"{cat}: {val}") clients_list.append({ "id": f"c{i}", "name": str(row['Customer']).strip().replace('\n', ' '), "priority": str(row['Active']).strip() if pd.notna(row['Active']) else 'x', "scope": scope, "notes": str(row['How can we be more proactive']).strip() if pd.notna(row['How can we be more proactive']) else '' }) # Output the full list to use in the code print(json.dumps(clients_list))
